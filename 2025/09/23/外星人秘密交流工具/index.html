    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Text Encoder/Decoder</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            max-width: 600px;
            margin: 0 auto;
            padding: 20px;
        }
        textarea {
            width: 100%;

            padding: 10px;
            margin-right: 10px;
        }
        #result {
            margin-top: 20px;
            padding: 10px;
            border: 1px solid #ccc;
            min-height: 100px;
        }
    </style>
</head>
<body>
    <h1>Text Encoder/Decoder</h1>
    <textarea id="inputText" placeholder="Enter text here..."></textarea>
    <div>
        <button onclick="encodeText()">Encode</button>
        <button onclick="decodeText()">Decode</button>
    </div>
    <div id="result"></div>

    <script>
        function encodeText() {
            const input = document.getElementById('inputText').value;
            if (!input) {
                alert('请输入需要加密的文本！');
                return;
            }

            // 将中文字符转换为UTF-16编码（4位十六进制）
            let hexString = '';
            for (let i = 0; i < input.length; i++) {
                const code = input.charCodeAt(i);
                hexString += code.toString(16).padStart(4, '0');
            }

            // 严格映射十六进制字符到“啦咯噜咧”
            const map = {
                '0': '咯咯', '1': '咯啦', '2': '咯噜', '3': '咯咧',
                '4': '啦咯', '5': '啦啦', '6': '啦噜', '7': '啦咧',
                '8': '噜咯', '9': '噜啦', 'a': '噜噜', 'b': '噜咧',
                'c': '咧咯', 'd': '咧啦', 'e': '咧噜', 'f': '咧咧'
            };

            let encodedText = '';
            for (const char of hexString) {
                encodedText += map[char] || '';
            }

            document.getElementById('result').textContent = encodedText;
        }

        function decodeText() {
            const input = document.getElementById('inputText').value;
            if (!input) {
                alert('请输入需要解密的文本！');
                return;
            }

            // 检查输入是否仅包含“啦”、“咯”、“噜”、“咧”
            if (!/^[啦咯噜咧]+$/.test(input)) {
                alert('加密内容包含非法字符！');
                return;
            }

            // 确保输入长度为偶数
            if (input.length % 2 !== 0) {
                alert('加密内容长度不正确！');
                return;
            }

            // 反向映射“啦咯噜咧”到十六进制字符
            const map = {
                '咯咯': '0', '咯啦': '1', '咯噜': '2', '咯咧': '3',
                '啦咯': '4', '啦啦': '5', '啦噜': '6', '啦咧': '7',
                '噜咯': '8', '噜啦': '9', '噜噜': 'a', '噜咧': 'b',
                '咧咯': 'c', '咧啦': 'd', '咧噜': 'e', '咧咧': 'f'
            };

            // 将加密字符串转换为十六进制
            let hexString = '';
            for (let i = 0; i < input.length; i += 2) {
                const pair = input.substr(i, 2);
                hexString += map[pair] || '';
            }

            // 将十六进制字符串还原为原始文本
            let decodedText = '';
            for (let i = 0; i < hexString.length; i += 4) {
                const hex = hexString.substr(i, 4);
                if (hex.length === 4) {
                    const charCode = parseInt(hex, 16);
                    if (!isNaN(charCode)) {
                        decodedText += String.fromCharCode(charCode);
                    } else {
                        console.error(`Invalid hex code: ${hex}`);
                    }
                }
            }

            document.getElementById('result').textContent = decodedText;
        }
    </script>
</body>
</html>

一个简单的文本编码器，采用了 UTF-16 编码原理用“啦咯噜咧”这四个字代替了十六进制数字
